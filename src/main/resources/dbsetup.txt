-- Drop tables if they exist
DROP TABLE IF EXISTS media_content, media, users;

-- Table to store users
CREATE TABLE users (
    user_id INTEGER PRIMARY KEY,
    username VARCHAR(255) NOT NULL ,
    password VARCHAR(255) NOT NULL,
    shared_symmetric_key VARCHAR(255) NOT NULL,
    family_symmetric_key VARCHAR(255) NOT NULL,
    UNIQUE(shared_symmetric_key),
    UNIQUE(family_symmetric_key)
);

-- Table to store media information
CREATE TABLE media (
    title VARCHAR(255) PRIMARY KEY,
    owner_id INTEGER REFERENCES users(user_id) ON DELETE CASCADE,
    format VARCHAR(50) NOT NULL,
    artist VARCHAR(255) NOT NULL,
    genre VARCHAR(255) NOT NULL
);

-- Table to store media content
CREATE TABLE media_content (
    title_content VARCHAR(255) PRIMARY KEY,
    lyrics TEXT NOT NULL,
    file_path TEXT NOT NULL,
    FOREIGN KEY (title_content) REFERENCES media(title) ON DELETE CASCADE
);
